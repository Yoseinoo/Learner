Table user {
  id integer [primary key]
  email varchar
  password varchar
  cardNumber integer
}

Table card {
  id integer [primary key]
  question varchar
  answer varchar
  level integer
  deck_id integer
}

Table deck {
  id integer [primary key]
  label varchar
  description varchar
  active boolean
  public boolean
  category_id integer
  user_id integer
}

Table category {
  id integer [primary key]
  label varchar
}

Ref: deck.category_id > category.id
Ref: deck.user_id > user.id

Ref: card.deck_id > deck.id
